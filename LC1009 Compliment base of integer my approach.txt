class Solution {
public:
    int bitwiseComplement(int n) {
        int x =n;int c=0;int d,r;
        r=0;
        if(n==0)
         r=1;
        while(x!=0){
           x= x>>1;
            c++;
        }
        x=n;
       for(int i=1;i<=c;i++){
        if (x&1){
            //digit 1 tha
            d=0;
        }
        else{
            d=1;
        }
        r+= d*pow(2,i-1);


        x=x>>1;

       }
       return r; 

    }
};
